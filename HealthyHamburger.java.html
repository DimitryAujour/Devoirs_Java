package Burgers;

public class HealthyHamburger extends Hamburger {
    public HealthyHamburger(String meat, double price) {
        super("Brown rye", meat, price);
    }

    @Override
    // Verifies if the addition is valid and adds it to the additions list
    public void addAddition(String addition) {
        if (additions.size() < 6) {
            additions.add(addition);
            System.out.println("Added " + addition + " for an extra " + getAdditionPrice(addition));
        } else {
            System.out.println("Cannot add more than 6 additional items to a healthy burger.");
        }
    }
}
