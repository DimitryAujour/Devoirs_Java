package Burgers;


import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

public class Hamburger {
    private String rollType;
    private String meat;
    private double price;
    public List<String> additions;

    //The constructor initializes the fields
    public Hamburger(String rollType, String meat, double price) {
        this.rollType = rollType;
        this.meat = meat;
        this.price = price;
        this.additions = new ArrayList<>();
    }
// Verifies if the addition is valid and adds it to the additions list
    public void addAddition(String addition) {
        if (additions.size() < 4) {
            additions.add(addition);
            System.out.println("Added " + addition + " for an extra " + getAdditionPrice(addition));
        } else {
            System.out.println("Cannot add more than 4 additional items.");
        }
    }

    public double getAdditionPrice(String addition) {
        // Switch statement to get the price of the addition
        double additionPrice = 0.0;
        switch (addition.toLowerCase()) {
            case "lettuce":
                additionPrice += 0.5;
                break;
            case "tomato":
                additionPrice += 0.5;
                break;
            case "cheese":
                additionPrice += 1;
                break;
            case "bacon":
                additionPrice += 1.5;
                break;
            default:
                System.out.println("Invalid addition");
                break;
        }
        return additionPrice;
    }

    public double calculateTotalPrice() {
        double totalPrice = price;
        for (String addition : additions) {
            totalPrice += getAdditionPrice(addition);
        }
        return totalPrice;
    }
// Displays the order
    public void displayOrder() {
        System.out.println("Hamburger on a " + rollType + " roll with " + meat + ", price is " + price);
        for (String addition : additions) {
            System.out.println("Added " + addition + " for an extra " + getAdditionPrice(addition));
        }
        System.out.println("Total Burger price is " + calculateTotalPrice());
    }
}
